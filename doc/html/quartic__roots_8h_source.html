41 inline void quartic_roots(const complex &a, const complex &b, const complex &c,
42  const complex &d, const complex &e,
43  complex &x1, complex &x2, complex &x3, complex &x4)
44 {
45  // Case of a biquadratic form
46  if ((std::norm(b) == 0.0) && (std::norm(d) == 0.0)) // Case b = d = 0
47  {
48  // Solving of the quadratic roots
49  complex zp, zm;
50  quadratic_roots(a, c, e, zp, zm);
51  // Finding the quartic roots
52  complex sqrtzp = Comparison::true_zero(std::sqrt(zp)),
53  sqrtzm = Comparison::true_zero(std::sqrt(zm));
54  x1 = sqrtzm;
55  x2 = -sqrtzm;
56  x3 = sqrtzp;
57  x4 = -sqrtzp;
58  }
59  else
60  {
61  if (std::norm(a) > 0.0) // a != 0
62  {
63  // Note: 3/8 = 0.375, 1/4 = 0.25, 1/8 = 0.125, 1/16 = 0.0625, 1/64 = 0.015625, 3/256 = 0.01171875
64  // Conversion to a depressed quartic
65  complex inva = 1.0 / a;
66  complex nb = b * inva, nc = c * inva, nd = d * inva, ne = e * inva;
67  complex nb2 = nb * nb; // Precompute reusable values
68  complex f = nc - 0.375 * nb2,
69  g = nd + 0.125 * nb2 * nb - 0.5 * nb * nc,
70  h = ne - 0.01171875 * nb2 * nb2 + 0.0625 * nb2 * nc - 0.25 * nb * nd;
71  // Computation of the roots of the cubic resolvent
72  complex r1, r2, r3;
73  cubic_roots(1.0, 0.5 * f, 0.0625 * (f * f - 4.0 * h), -0.015625 * g * g, r1, r2, r3);
74  // We take the two complex roots (if complex), that is r2 and r3. If one of them
75  // is zero, we take r1
76  complex p, q;
77  if (!Comparison::almost_zero(r2))
78  {
79  p = std::sqrt(r2);
80  if (!Comparison::almost_zero(r3))
81  q = std::sqrt(r3); // p = sqrt(r2), q = sqrt(r3)
82  else
83  q = std::sqrt(r1); // p = sqrt(r2), q = sqrt(r1)
84  }
85  else // p = sqrt(r1), q = sqrt(r3)
86  {
87  p = std::sqrt(r1);
88  q = std::sqrt(r3);
89  }
90  //
91  complex r = -0.125 * g / (p * q),
92  s = 0.25 * nb;
93  x1 = Comparison::true_zero(p + q + r - s);
94  x2 = Comparison::true_zero(p - q - r - s);
95  x3 = Comparison::true_zero(-p + q - r - s);
96  x4 = Comparison::true_zero(-p - q + r - s);
97  }
98  else

101  }
