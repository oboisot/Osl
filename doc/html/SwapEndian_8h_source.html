25  static inline std::uint16_t bswap16(const std::uint16_t &x)
26  {
27  return (x << 8) | (x >> 8);
28  }
29  static inline std::uint32_t bswap32(const std::uint32_t &x)
30  {
31  std::uint32_t step16 = (x << 16) | (x >> 16);
32  return ((step16 << 8) & 0xff00ff00) | ((step16 >> 8) & 0x00ff00ff);
33  }
34  static inline std::uint64_t bswap64(const std::uint64_t &x)
35  {
36  std::uint64_t step32 = (x << 32) | (x >>32);
37  std::uint64_t step16 = ((step32 & 0x0000FFFF0000FFFFULL) << 16) | ((step32 & 0xFFFF0000FFFF0000ULL) >> 16);
38  return ((step16 & 0x00FF00FF00FF00FFULL) << 8) | ((step16 & 0xFF00FF00FF00FF00ULL) >> 8);
39  }

121 {
