60  complex &x1, complex &x2, complex &x3)
61 {
62  if (std::norm(a) > 0.0) // if |a|^2 > 0 => a != 0
63  {
64  complex a2 = a * a, b2 = b * b, ac = a * c; // Precompute reusable values
65  complex Q = (3.0 * ac - b2) / (9.0 * a2),
66  R = (9.0 * ac * b - 27.0 * a2 * d - 2.0 * b2 * b) / (54.0 * a2 * a);
67  complex sqrtD = std::sqrt(Q * Q * Q + R * R);
68  complex S = cbrt(R + sqrtD),
69  U = cbrt(R - sqrtD);
70  complex s0 = -b / (3.0 * a),
71  s1 = -0.5 * (S + U),
72  s2 = complex(0, 0.5 * Constants::m_sqrt_3) * (S - U); // i*sqrt(3)/2 * (S - U)
73  // Cubic roots
74  x1 = Comparison::true_zero(S + U + s0); // Should be the real solution
75  x2 = Comparison::true_zero(s1 + s2 + s0);
76  x3 = Comparison::true_zero(s1 - s2 + s0);
77  }
